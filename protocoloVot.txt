protocolo-votacion = 1*(mensaje-cliente / mensaje-servidor)

mensaje-cliente = comando-votar / comando-cerrar
comando-votar   = "VOTAR" SP dni SP opcion CRLF
comando-cerrar  = "CERRAR" CRLF

mensaje-servidor = respuesta-ok / respuesta-error / respuesta-final
respuesta-ok    = "200" SP "OK" CRLF
respuesta-error = cod-error SP motivo CRLF
respuesta-final = "201" SP "Ganador:" SP candidato SP "Recuento final:" SP recuento CRLF

dni             = 8DIGIT %x41-5A
opcion          = "ANDREA_MARTOS" / "JAVIER_GARCIA" / "PEDRO_GOMEZ"
candidato       = opcion
cod-error       = "601" / "403" / "404"
motivo          = "YA_VOTADO" / "ERROR_CERRADO" / "OPCION_INVALIDA"
recuento        = "{" 1*(candidato ":" 1*DIGIT [", "]) "}"

DIGIT           = %x30-39
SP              = %x20
CRLF            = %x0D %x0A
